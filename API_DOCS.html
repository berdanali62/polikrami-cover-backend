🔐 Kimlik Doğrulama (Authentication)
POST Kayıt Olma
/api/v1/auth/register
Kimlik Doğrulama:
Gerekli değil
Fonksiyon:
registerController()
Request
{
  email: string;              // Geçerli email formatı
  password: string;           // Min 8 karakter, en az 1 büyük harf, 1 küçük harf, 1 rakam, 1 özel karakter
  confirmPassword: string;    // password ile aynı olmalı
  name?: string;             // Min 2, Max 100 karakter (opsiyonel)
  role?: "user" | "designer"; // Default: "user"
  acceptTerms: true;         // Zorunlu - true olmalı
  acceptPrivacy: true;       // Zorunlu - true olmalı
  acceptRevenueShare?: true; // Tasarımcılar için zorunlu
}
Response
{
  user: {
    id: string;
    email: string;
    name: string | null;
  };
}
